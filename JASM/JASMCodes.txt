NOP = No Operation
NTK <label> = New Track 
PRC <program> = Program Change
BCN <bank> = Bank Change
PRF <type>,<value>,<duration> = perf (controls stuff like pitch bend)
NON <voice>,<note>,<velocity>  = note on
NOF <voice> = voice off.
TBS <bpm>,<ppqn> = set timebase. 
PAR <parameter>,<value> = Sequencer parameter
DEL <ticks> = delay 
JMP <label> = jump
